    50→    50→// ── Image Crop Dialog Component ──
    51→    51→function ImageCropDialog({
    52→    52→  imageSrc,
    53→    53→  onCropComplete,
    54→    54→  onCancel,
    55→    55→  aspectRatio = 1,
    56→    56→  t,
    57→    57→}: {
    58→    58→  imageSrc: string;
    59→    59→  onCropComplete: (croppedBlob: Blob) => void;
    60→    60→  onCancel: () => void;
    61→    61→  aspectRatio?: number;
    62→    62→  t: (ar: string, en: string) => string;
    63→    63→}) {
    64→    64→  const [crop, setCrop] = useState<CropType>();
    65→    65→  const [completedCrop, setCompletedCrop] = useState<PixelCrop>();
    66→    66→  const imgRef = useRef<HTMLImageElement>(null);
    67→    67→  const [processing, setProcessing] = useState(false);
    68→    68→
    69→    69→  const handleCropAndSave = useCallback(() => {
    70→    70→    if (!completedCrop || !imgRef.current) return;
    71→    71→    setProcessing(true);
    72→    72→
    73→    73→    const image = imgRef.current;
    74→    74→    const canvas = document.createElement('canvas');
    75→    75→    const scaleX = image.naturalWidth / image.width;
    76→    76→    const scaleY = image.naturalHeight / image.height;
    77→    77→    const cropX = completedCrop.x * scaleX;
    78→    78→    const cropY = completedCrop.y * scaleY;
    79→    79→    const cropWidth = completedCrop.width * scaleX;
    80→    80→    const cropHeight = completedCrop.height * scaleY;
    81→    81→
    82→    82→    canvas.width = cropWidth;
    83→    83→    canvas.height = cropHeight;
    84→    84→    const ctx = canvas.getContext('2d');
    85→    85→    if (!ctx) return;
    86→    86→
    87→    87→    ctx.drawImage(
    88→    88→      image,
    89→    89→      cropX,
    90→    90→      cropY,
    91→    91→      cropWidth,
    92→    92→      cropHeight,
    93→    93→      0,
    94→    94→      0,
    95→    95→      cropWidth,
    96→    96→      cropHeight
    97→    97→    );
    98→    98→
    99→    99→    canvas.toBlob(
   100→   100→      (blob) => {
   101→   101→        if (blob) {
   102→   102→          onCropComplete(blob);
   103→   103→        }
   104→   104→        setProcessing(false);
   105→   105→      },
   106→   106→      'image/jpeg',
   107→   107→      0.92
   108→   108→    );
   109→   109→  }, [completedCrop, onCropComplete]);
   110→   110→
   111→   111→  return (
   112→   112→    <div className="fixed inset-0 z-[200] flex items-center justify-center bg-black/80 backdrop-blur-sm p-4">
   113→   113→      <div className="w-full max-w-lg rounded-2xl border border-white/10 bg-[#0B2545] p-6 shadow-2xl">
   114→   114→        <div className="flex items-center justify-between mb-4">
   115→   115→          <h3 className="text-lg font-bold text-white">
   116→   116→            {t('قص الصورة', 'Crop Image')}
   117→   117→          </h3>
   118→   118→          <button onClick={onCancel} className="text-white/50 hover:text-white cursor-pointer">
   119→   119→            <X className="h-5 w-5" />
   120→   120→          </button>
   121→   121→        </div>
   122→   122→        <div className="max-h-[60vh] overflow-auto rounded-lg bg-black/20 mb-4">
   123→   123→          <ReactCrop
   124→   124→            crop={crop}
   125→   125→            onChange={(c) => setCrop(c)}
   126→   126→            onComplete={(c) => setCompletedCrop(c)}
   127→   127→            aspect={aspectRatio}
   128→   128→          >
   129→   129→            <img
   130→   130→              ref={imgRef}
   131→   131→              src={imageSrc}
   132→   132→              alt="Crop"
   133→   133→              className="max-w-full"
   134→   134→              style={{ maxHeight: '50vh' }}
   135→   135→            />
   136→   136→          </ReactCrop>
   137→   137→        </div>
   138→   138→        <div className="flex items-center justify-end gap-3">
   139→   139→          <Button
   140→   140→            onClick={onCancel}
   141→   141→            variant="ghost"
   142→   142→            className="text-white/50 hover:text-white hover:bg-white/10"
   143→   143→          >
   144→   144→            {t('إلغاء', 'Cancel')}
   145→   145→          </Button>
   146→   146→          <Button
   147→   147→            onClick={handleCropAndSave}
   148→   148→            disabled={!completedCrop || processing}
   149→   149→            className="bg-gradient-to-r from-gold to-gold-light text-navy-900 hover:opacity-90 font-bold"
   150→   150→          >
   151→   151→            {processing ? (
   152→   152→              <Loader2 className="h-4 w-4 animate-spin" />
   153→   153→            ) : (
   154→   154→              <Crop className="h-4 w-4 mr-2 ml-2" />
   155→   155→            )}
   156→   156→            {t('قص وحفظ', 'Crop & Save')}
   157→   157→          </Button>
   158→   158→        </div>
   159→   159→      </div>
   160→   160→    </div>
   161→   161→  );
   162→   162→}
